28 int HID_::getInterface(uint8_t* interfaceCount) {
29  *interfaceCount += 1; // uses 1
30  HIDDescriptor hidInterface = {
31  D_INTERFACE(pluggedInterface, 1, USB_DEVICE_CLASS_HUMAN_INTERFACE, HID_SUBCLASS_NONE, HID_PROTOCOL_NONE),

38 int HID_::getDescriptor(USBSetup& setup) {
39  // Check if this is a HID Class Descriptor request
40  if (setup.bmRequestType != REQUEST_DEVICETOHOST_STANDARD_INTERFACE) {
41  return 0;
42  }
43  if (setup.wValueH != HID_REPORT_DESCRIPTOR_TYPE) {
44  return 0;
45  }
46 
47  // In a HID Class Descriptor wIndex cointains the interface number
48  if (setup.wIndex != pluggedInterface) {
49  return 0;
50  }
51 
52  int total = 0;
53  HIDSubDescriptor* node;
54  for (node = rootNode; node; node = node->next) {
55  int res = USB_SendControl(TRANSFER_PGM, node->data, node->length);
56  if (res == -1)
57  return -1;
58  total += res;
59  }
60 
61  // Reset the protocol on reenumeration. Normally the host should not assume the state of the protocol
62  // due to the USB specs, but Windows and Linux just assumes its in report mode.
63  protocol = HID_REPORT_PROTOCOL;
64 
65  return total;
66 }

78 void HID_::AppendDescriptor(HIDSubDescriptor *node) {
79  if (!rootNode) {
80  rootNode = node;

82  HIDSubDescriptor *current = rootNode;
83  while (current->next) {

91 int HID_::SendReport(uint8_t id, const void* data, int len) {
92  auto ret = USB_Send(pluggedEndpoint, &id, 1);
93  if (ret < 0) return ret;
94  auto ret2 = USB_Send(pluggedEndpoint | TRANSFER_RELEASE, data, len);
95  if (ret2 < 0) return ret2;
96  return ret + ret2;
97 }
