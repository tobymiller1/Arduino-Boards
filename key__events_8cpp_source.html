46 void handleKeyswitchEvent(Key mappedKey, byte row, byte col, uint8_t keyState) {
47  /* These first steps are only done for keypresses that have a real (row,col).
48  * In particular, doing them for keypresses with out-of-bounds (row,col)
49  * would cause out-of-bounds array accesses in Layer.lookup(),
50  * Layer.updateLiveCompositeKeymap(), etc.
51  * Perhaps this check should instead be for INJECTED - I'm not clear on
52  * whether keypresses with out-of-bounds (row,col) are the same as
53  * keypresses with INJECTED set, or are a superset or a subset of
54  * INJECTED keypresses. In any case, the (row,col) test is *safe* in that
55  * it avoids out-of-bounds accesses, at least in core. (Can't promise
56  * anything about event handlers - they may still receive out-of-bounds
57  * (row,col), and handling that properly is on them.)
58  */
59  if (row < ROWS && col < COLS) {
60 
61  /* If a key had an on or off event, we update the live composite keymap. See
62  * layers.h for an explanation about the different caches we have. */
63  if (keyToggledOn(keyState) || keyToggledOff(keyState))
64  Layer.updateLiveCompositeKeymap(row, col);
65 
66  /* If the key we are dealing with is masked, ignore it until it is released.
67  * When releasing it, clear the mask, so future key events can be handled
68  * appropriately.
69  *
70  * See layers.cpp for an example that masks keys, and the reason why it does
71  * so.
72  */
73  if (KeyboardHardware.isKeyMasked(row, col)) {
74  if (keyToggledOff(keyState)) {
75  KeyboardHardware.unMaskKey(row, col);
76  } else {
77  return;
78  }
79  }
80 
81  if (!(keyState & INJECTED)) {
82  mappedKey = Layer.lookup(row, col);
83  }
84 
85  } // row < ROWS && col < COLS
86 
87  // Keypresses with out-of-bounds (row,col) start here in the processing chain
88 
89  for (byte i = 0; Kaleidoscope.eventHandlers[i] != NULL && i < HOOK_MAX; i++) {
90  Kaleidoscope_::eventHandlerHook handler = Kaleidoscope.eventHandlers[i];
91  mappedKey = (*handler)(mappedKey, row, col, keyState);
92  if (mappedKey.raw == Key_NoKey.raw)
93  return;
94  }
95  mappedKey = Layer.eventHandler(mappedKey, row, col, keyState);
96  if (mappedKey.raw == Key_NoKey.raw)
97  return;
98  handleKeyswitchEventDefault(mappedKey, row, col, keyState);
99 }
