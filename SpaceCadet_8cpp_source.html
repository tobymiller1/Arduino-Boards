62 Key SpaceCadet::eventHandlerHook(Key mapped_key, byte row, byte col, uint8_t key_state) {
63 
64  // If nothing happened, bail out fast.
65  if (!keyIsPressed(key_state) && !keyWasPressed(key_state)) {
66  return mapped_key;
67  }
68 
69  // If a key has been just toggled on...
70  if (keyToggledOn(key_state)) {
71 
72  //This will only set one key, and if it isn't in our map it clears everything
73  //for the non-pressed key
74  //Exit condition is if we reach the sentinal
75  for (
76  uint8_t i = 0 ;
77  !(
78  map[i].input.raw == Key_NoKey.raw
79  && map[i].output.raw == Key_NoKey.raw
80  && map[i].timeout == 0
81  ) ;
82  ++i
83  ) {
84 
85  if (mapped_key.raw == map[i].input.raw) {
86  //The keypress was valid and a match.
87  map[i].flagged = true;
88  map[i].start_time = millis();
89  } else {
90  //The keypress wasn't a match.
91  map[i].flagged = false;
92  map[i].start_time = 0;
93  }
94  }
95 
96  // this is all we need to do on keypress, let the next handler do its thing too.
97  return mapped_key;
98  }
99 
100  // if the state is empty, that means that either the shifts weren't pressed,
101  // or we used another key in the interim. in both cases, nothing special to do.
102  bool valid_key = false;
103  bool pressed_key_was_valid = false;
104  uint8_t index = 0;
105 
106  //Look to see if any keys in our map are flagged.
107  //Exit condition is if we reach the sentinal
108  for (
109  uint8_t i = 0 ;
110  !(
111  map[i].input.raw == Key_NoKey.raw
112  && map[i].output.raw == Key_NoKey.raw
113  && map[i].timeout == 0
114  ) ;
115  ++i
116  ) {
117 
118  if (map[i].flagged) {
119  valid_key = true;
120  index = i;
121  }
122  if (map[i].input.raw == mapped_key.raw) {
123  pressed_key_was_valid = true;
124  }
125  }
126 
127 
128  //If no valid mapped keys were pressed, simply return the keycode
129  if (!valid_key) {
130  return mapped_key;
131  }
132 
133  //use the map index to find the local timeout for this key
134  uint16_t current_timeout = map[index].timeout;
135  //If that isn't set, use the global timeout setting.
136  if (current_timeout == 0) {
137  current_timeout = time_out;
138  }
139 
140  if ((millis() - map[index].start_time) >= current_timeout) {
141  // if we timed out, that means we need to keep pressing the mapped
142  // key, but we won't need to send the alternative key in the end
143  map[index].flagged = false;
144  map[index].start_time = 0;
145  return mapped_key;
146  }
147 
148  // If the key that was pressed isn't one of our mapped keys, just
149  // return. This can happen when another key is released, and that should not
150  // interrupt us.
151 
152  if (!pressed_key_was_valid) {
153  return mapped_key;
154  }
155 
156  // if a key toggled off (and that must be one of the mapped keys at this point),
157  // send the alternative key instead (if we were interrupted, we bailed out earlier).
158  if (keyToggledOff(key_state)) {
159  Key alternate_key = map[index].output;
160 
161  //Since we are sending the actual key (no need for shift, etc),
162  //only need to send that key and not the original key. In fact, we
163  //may want to even UNSET the originally pressed key (future
164  //enhanacement?). This might also mean we don't need to return the
165  //key that was pressed, though I haven't confirmed that.
166  handleKeyswitchEvent(alternate_key, row, col, IS_PRESSED | INJECTED);
167  hid::sendKeyboardReport();
168 
169  //Unflag the key so we don't try this again.
170  map[index].flagged = false;
171  map[index].start_time = 0;
172  }
173 
174  return mapped_key;
175 }

177 }
