27 static const uint8_t _hidMultiReportDescriptorConsumer[] PROGMEM = {

64 void ConsumerControl_::press(uint16_t m) {
65  // search for a free spot
66  for (uint8_t i = 0; i < sizeof(HID_ConsumerControlReport_Data_t) / 2; i++) {
67  if (_report.keys[i] == 0x00) {
68  _report.keys[i] = m;
69  break;
70  }
71  }
72  sendReport(&_report, sizeof(_report));
73 }
74 
75 void ConsumerControl_::release(uint16_t m) {
76  // search and release the keypress
77  for (uint8_t i = 0; i < sizeof(HID_ConsumerControlReport_Data_t) / 2; i++) {
78  if (_report.keys[i] == m) {
79  _report.keys[i] = 0x00;
80  // no break to delete multiple keys
81  }
82  }
83  sendReport(&_report, sizeof(_report));
84 }
